简单版


# 1.打包
npm run build

# 2.上传（使用xshell上传dist目录）
rz

# 3.解压缩并且删除没用的压缩文件
unzip dist.zip
rm -rf dist.zip

# 4.部署
docker run -p 80:80 -v $PWD/dist:/usr/share/nginx/html -d nginx
